73 function compare_terms(p1::AbstractPolynomial, p2::AbstractPolynomial, isz, op)
74 i1 = 1
75 i2 = 1
76 t1 = terms(p1)
77 t2 = terms(p2)
78 while true
79 while i1 <= length(t1) && isz(coefficient(t1[i1]))
80 i1 += 1
81 end
82 while i2 <= length(t2) && isz(coefficient(t2[i2]))
83 i2 += 1
84 end
85 if i1 > length(t1) && i2 > length(t2)
86 return true
87 end
88 if i1 > length(t1) || i2 > length(t2)
89 return false
90 end
91 if !op(t1[i1], t2[i2])
92 return false
93 end
94 i1 += 1
95 i2 += 1
96 end

364

366

368
369 The following example shows how to generate all exponents of
370 monomials of 2 variables up to degree 2.
371 ```jldoctest
372 julia> collect(ExponentsIterator{Graded{LexOrder}}((0, 0), maxdegree = 2))
373 6-element Vector{Tuple{Int64, Int64}}:

375 (0, 1)
376 (1, 0)
377 (0, 2)
378 (1, 1)
379 (2, 0)
380 ```

382 of arbitrary length using `ntuple` as follows:
383 ```jldoctest
384 julia> collect(ExponentsIterator{Graded{LexOrder}}(ntuple(zero, 3), mindegree = 2, maxdegree = 2))

386 (0, 0, 2)
387 (0, 1, 1)
388 (0, 2, 0)
389 (1, 0, 1)
390 (1, 1, 0)
391 (2, 0, 0)
392 ```
393 You can also change the monomial ordering and use `Vector` instead of `Tuple` as follows:
394 ```jldoctest
395 julia> collect(ExponentsIterator{LexOrder}(zeros(Int, 2), mindegree = 2, maxdegree = 3))
396 7-element Vector{Vector{Int64}}:
